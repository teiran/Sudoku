6   package kayttoliityma;
7   
8   import java.awt.Container;
9   import java.awt.Dimension;
10  import java.awt.GridLayout;
11  import javax.swing.JFrame;
12  import javax.swing.WindowConstants;
13  import sovellusLogiikka.Ruudukko;
14  
15  /**
16   *
17   * @author tiera
18   */
19  public class Kayttoliityma implements Runnable {
20  
21      private JFrame frame;
22      private final Ruudukko taulukko;
23      private TarkistusButton tarkista;
24  
25      public Kayttoliityma(Ruudukko r) {
26          this.taulukko = r;
27      }
28  
29      @Override
30      public void run() {
31          frame = new JFrame("Sudoku");
32          frame.setPreferredSize(new Dimension(1000, 1000));
33  
34          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
35  
36          luokomponentti(frame.getContentPane());
37  
38          frame.pack();
39          frame.setVisible(true);
40  
41      }
42  
43      private void luokomponentti(Container container) {
44          frame.setLayout(new GridLayout(10, 10));
45          for (int i = 0; i < taulukko.getKorkeus(); i++) {
46              for (int j = 0; j < taulukko.getLeveys(); j++) {
47                  RuutuButton s = taulukko.getRuudukko()[i][j];
48                  s.addKeyListener(new SuodokuruudunKeyKuuntelija(s));
49                  SudokuruudunKuutelija n = new SudokuruudunKuutelija(s);
50                  s.addActionListener(n);
51                  container.add(s);
52              }
53  
54          }
55  
56          tarkista = new TarkistusButton(taulukko);
57          tarkista.addActionListener(new Tarkistalisener(tarkista));
58          container.add(tarkista);
59  
60      }
61  
62  }
