45      private void lisaanumerov(int t, int y, int x) {
46          valmiskartta[x][y] = t;

49      private void lisaanumeroa(int t, int y, int x) {
50          aloitukartta[x][y] = t;

53      private void lisaanumero(int t, int x, int y, boolean onko) {
54          if (onko) {
55              lisaanumeroa(t, x, y);
56          }
57          lisaanumerov(t, x, y);
58      }

60      private void lisaa(String k) {
